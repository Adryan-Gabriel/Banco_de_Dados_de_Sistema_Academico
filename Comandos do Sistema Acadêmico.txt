--Comandos do Sistema Acadêmico

-- 1. Usuario
CREATE TABLE Usuario (
    ID_Usuario INT PRIMARY KEY,
    NOME VARCHAR(100),
    CPF VARCHAR(14) NOT NULL UNIQUE,
    Telefone VARCHAR(15), 
    E_MAIL VARCHAR(100) NOT NULL, 
    SENHA VARCHAR(100) NOT NULL
);

-- 2. Professor
CREATE TABLE Professor (
    ID_Professor INT PRIMARY KEY,
    FORMAÇÃO VARCHAR(100) NOT NULL,
    SALÁRIO DECIMAL(10,2) NOT NULL,
    ID_Usuario INT NOT NULL,
    FOREIGN KEY (ID_Usuario) REFERENCES Usuario(ID_Usuario) ON DELETE CASCADE
);

-- 3. Aluno 
CREATE TABLE Aluno (
    ID_Aluno INT PRIMARY KEY,
    ID_Usuario INT NOT NULL,
    FOREIGN KEY (ID_Usuario) REFERENCES Usuario(ID_Usuario) ON DELETE CASCADE
);

-- 4. Curso
CREATE TABLE Curso (
    ID_Curso INT PRIMARY KEY,
    MODALIDADE VARCHAR(50) NOT NULL,
    DURAÇÃO VARCHAR(50) NOT NULL
);

-- 5. Grade
CREATE TABLE Grade (
    ID_Grade INT PRIMARY KEY,
    ID_Curso INT,
    CARGA_HORÁRIA_TOTAL INT NOT NULL,
    FOREIGN KEY (ID_Curso) REFERENCES Curso(ID_Curso) ON DELETE CASCADE
);

-- 6. Disciplina
CREATE TABLE Disciplina (
    ID_Disciplina INT PRIMARY KEY,
    ID_Grade INT,
    nome VARCHAR(100) NOT NULL,
    creditos INT NOT NULL,
    carga_horaria INT NOT NULL,
    FOREIGN KEY (ID_Grade) REFERENCES Grade(ID_Grade) ON DELETE CASCADE
);

-- 7. Turma 
CREATE TABLE Turma (
    ID_Turma INT PRIMARY KEY,
    ID_Disciplina INT,
    ID_Professor INT,
    FOREIGN KEY (ID_Disciplina) REFERENCES Disciplina(ID_Disciplina) ON DELETE CASCADE,
    FOREIGN KEY (ID_Professor) REFERENCES Professor(ID_Professor) ON DELETE CASCADE
);

-- 8. Matricula
CREATE TABLE Matricula (
    ID_Matricula INT PRIMARY KEY,
    ID_Aluno INT,
    ID_Curso INT,
    SITUAÇÃO VARCHAR(30) NOT NULL,
    Ano_Semestre VARCHAR(10) NOT NULL,
    FOREIGN KEY (ID_Aluno) REFERENCES Aluno(ID_Aluno) ON DELETE CASCADE,
    FOREIGN KEY (ID_Curso) REFERENCES Curso(ID_Curso) ON DELETE CASCADE
);

-- 9. Aluno_Turma 
CREATE TABLE Aluno_Turma (
    ID_Aluno INT,
    ID_Turma INT,
    FREQUENCIA DECIMAL(5,2) NOT NULL,
    PRIMARY KEY (ID_Aluno, ID_Turma),
    FOREIGN KEY (ID_Aluno) REFERENCES Aluno(ID_Aluno) ON DELETE CASCADE,
    FOREIGN KEY (ID_Turma) REFERENCES Turma(ID_Turma) ON DELETE CASCADE
);

-- 10. Historico
CREATE TABLE Historico (
    ID_Historico INT PRIMARY KEY,
    ID_Aluno INT,
    ID_Turma INT,
    NOTA DECIMAL(4,2) NOT NULL,
    FREQUENCIA DECIMAL(5,2) NOT NULL,
    FOREIGN KEY (ID_Aluno) REFERENCES Aluno(ID_Aluno) ON DELETE CASCADE,
    FOREIGN KEY (ID_Turma) REFERENCES Turma(ID_Turma) ON DELETE CASCADE
);
